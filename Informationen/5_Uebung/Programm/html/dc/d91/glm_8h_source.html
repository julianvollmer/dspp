00141 /* glmVertexNormals: Generates smooth vertex normals for a model.
00142  * First builds a list of all the triangles each vertex is in.  Then
00143  * loops through each vertex in the the list averaging all the facet
00144  * normals of the triangles each vertex is in.  Finally, sets the
00145  * normal index in the triangle for the vertex to the generated smooth
00146  * normal.  If the dot product of a facet normal and the facet normal
00147  * associated with the first triangle in the list of triangles the
00148  * current vertex is in is greater than the cosine of the angle
00149  * parameter to the function, that facet normal is not added into the
00150  * average normal calculation and the corresponding vertex is given
00151  * the facet normal.  This tends to preserve hard edges.  The angle to
00152  * use depends on the model, but 90 degrees is usually a good start.
00153  *
00154  * model - initialized GLMmodel structure
00155  * angle - maximum angle (in degrees) to smooth across
00156  */
00157 GLvoid
00158 glmVertexNormals(GLMmodel* model, GLfloat angle);

00169 /* glmSpheremapTexture: Generates texture coordinates according to a
00170  * spherical projection of the texture map.  Sometimes referred to as
00171  * spheremap, or reflection map texture coordinates.  It generates
00172  * these by using the normal to calculate where that vertex would map
00173  * onto a sphere.  Since it is impossible to map something flat
00174  * perfectly onto something spherical, there is distortion at the
00175  * poles.  This particular implementation causes the poles along the X
00176  * axis to be distorted.
00177  *
00178  * model - pointer to initialized GLMmodel structure
00179  */
00180 GLvoid
